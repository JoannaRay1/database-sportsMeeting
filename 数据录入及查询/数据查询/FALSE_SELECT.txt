⑭男子4×100米接力前三名的学院编号和学院名称

SELECT CL_NO,CL_NAME(FALSE)
FROM COLLEGE
WHERE CL_NO=(SELECT CL_NO
FROM COLLEGE_TEAM
WHERE CT_NO=(SELECT CT_NO
FROM TEAM_PARTICIPATION
WHERE RANK<=3 AND 
CNO=(SELECT CNO
FROM COMPETITION
WHERE CNAME='男子4*100接力')));
