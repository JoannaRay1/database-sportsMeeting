①信息管理与工程学院的参赛运动员信息
SELECT *
FROM ATHLETE
WHERE CL_NO=(SELECT CL_NO
FROM COLLEGE
WHERE CL_NAME='信息管理与工程学院');

②男子跳远的比赛结果
SELECT *
FROM INDIVIDUAL_PARTICIPATION
WHERE CNO=(SELECT CNO
FROM COMPETITION
WHERE CNAME='男子跳远');

③所有计时裁判员的姓名和编号
SELECT RE_NO,RE_NAME
FROM REFERREE
WHERE PNO=(SELECT PNO
FROM POST
WHERE PNAME='计时裁判员');
④参赛学院的个数
SELECT CL_NO
FROM COLLEGE
WHERE EXISTS(SELECT *
FROM COLLEGE_TEAM
WHERE CL_NO=COLLEGE_TEAM.CL_NO);


SELECT COUNT(CL_NO)
FROM COLLEGE
WHERE CL_NO IN(SELECT CL_NO
FROM COLLEGE_TEAM);

⑤获得女子4×100米项目第一名的学院编号和学院名称
SELECT CL_NO,CL_NAME
FROM COLLEGE

FROM COLLEGE_TEAM

⑩参加男子4*100接力的运动员学号和姓名

⑪获得女子800米前三的运动员姓名和学号

SELECT COMPETITION.CNAME,INDIVIDUAL_PARTICIPATION.RANK,ANAME,ID
FROM ATHLETE,INDIVIDUAL_PARTICIPATION,COMPETITION

INDIVIDUAL_PARTICIPATION.RANK<=3 AND
INDIVIDUAL_PARTICIPATION.CNO=COMPETITION.CNO AND
COMPETITION.CNAME='女子800米'

⑬赛事类别名称为“TRACK_EVENT”的项目名称、规则编号和场地编号
SELECT CNAME,RU_NO,FIELD.FNO,ENAME
FROM COMPETITION,RULE,FIELD,EVENT_CATEGORY
WHERE COMPETITION.CNO=RULE.CNO AND
COMPETITION.FNO=FIELD.FNO AND
COMPETITION.ENO=EVENT_CATEGORY.ENO AND
EVENT_CATEGORY.ENAME='TRACK_EVENT';




⑭男子4×100米接力前三名的学院编号和学院名称
SELECT  COMPETITION.CNAME,TEAM_PARTICIPATION.RANK,COLLEGE.CL_NO,COLLEGE.CL_NAME
FROM COLLEGE,COLLEGE_TEAM,TEAM_PARTICIPATION,COMPETITION
WHERE COLLEGE.CL_NO=COLLEGE_TEAM.CL_NO AND
COLLEGE_TEAM.CT_NO=TEAM_PARTICIPATION.CT_NO AND
COMPETITION.CNO=TEAM_PARTICIPATION.CNO AND
TEAM_PARTICIPATION.RANK<=3 AND 
 COMPETITION.CNAME='男子4*100接力'
ORDER BY TEAM_PARTICIPATION.RANK;
